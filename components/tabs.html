# tabs-标签

选项卡切换组件。

# 组件注册


<script>
import {Tabs, TabsBody, TabsHead, TabsItem, TabsPane} from "aisaka"

export default {
  components: {
    "a-tabs": Tabs,
    "a-tabs-head": TabsHead,
    "a-tabs-body": TabsBody,
    "a-tabs-item": TabsItem,
    "a-tabs-pane": TabsPane
  }
}
</script>
